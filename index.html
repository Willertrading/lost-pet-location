        return;

      }


      status.textContent = "Requesting location…";


      navigator.geolocation.getCurrentPosition(position => {

        fetch('https://script.google.com/macros/s/AKfycbxyHp_fK-ah2lWIJ7XUbU0_Sekkcfebtz_gPC4jvLvT9ZdrXdJNJXXtk9NvlJ_jEZiAbw/exec

', {

          method: 'POST',

          headers: { 'Content-Type': 'application/json' },

          body: JSON.stringify({

            lat: position.coords.latitude,

            lon: position.coords.longitude

          })

        }).then(() => {

          status.textContent = "✅ Location sent! Thank you.";

        }).catch(err => {

        });

      }, () => {

      });
